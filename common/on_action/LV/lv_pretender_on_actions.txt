on_title_destroyed = {
	on_actions = {
		lv_pretender_empire_destroyed_on_action
	}
}

lv_pretender_empire_destroyed_on_action = { 
	trigger = {
		scope:landed_title = { has_variable = lv_pretender_empire }
	}
	effect = {
		every_character_war = {
			limit = {
				using_cb = dismantle_pretender_cb
				is_defender = prev
			}
			end_war = attacker
		}
	} 
}
