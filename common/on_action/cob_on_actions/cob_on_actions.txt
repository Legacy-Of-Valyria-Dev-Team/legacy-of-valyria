# called for armies whenever they enter a province
# root is the owner of the army
# scope:army is the army

on_travel_plan_arrival = {
	on_actions = {
		restore_valyria_cob_on_actions
	}
}

restore_valyria_cob_on_actions = {
 	effect = {
 		if = {
 			limit = {
 				character:Varezys_1 = {
 					AND = {
 						exists = yes
 						#is_ai = no
						location = province:8405	
 					}
 				}
 			}
 			character:Varezys_1 = {
 				trigger_event = lv_scenario_cob.0500
 			}
 		}
		if = {
 			limit = {
 				character:Varezys_1 = {
 					AND = {
 						exists = yes
 						#is_ai = no
						location = province:8194	
 					}
 				}
 			}
 			character:Varezys_1 = {
 				trigger_event = lv_scenario_cob.0501
 			}
 		}
 	}
}
