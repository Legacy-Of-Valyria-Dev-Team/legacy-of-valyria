		parochial_opinion = -20
		parochial_tax_contribution_mult = 0.1

				limit = {

					value = tradition_unfulfilled_criteria_penalty

	}

			limit = {

					scope:character = {

							percent >= 0.15

							}
						}
					}

			}

		}

					}
				}
			}
			multiply = 0
		}
	}
