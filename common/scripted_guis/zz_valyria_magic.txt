		scope:zz_magic_caster = {
			root = {
				clear_variable_list = zz_valyria_spell_list
				set_variable = {
					name = zz_magic_caster
					value = scope:zz_magic_caster
				}
			}
			scope:zz_magic_caster = {
				every_memory = {
					limit = { 
						has_memory_category = magic
					}

						add_to_variable_list = {
							name = zz_valyria_spell_list
							target = prev
						}
					}
				}
			}
		}
	}
