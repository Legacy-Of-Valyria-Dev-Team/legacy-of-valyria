zz_valyria_magic_teacher = {
	scope = character
	saved_scopes = {
		zz_magic_teacher
	}
	is_shown = {
		var:zz_magic_teacher ?= scope:zz_magic_teacher
	}
	effect = {
		scope:zz_magic_teacher = {
			root = {
				clear_variable_list = zz_valyria_spell_list
				remove_variable = zz_valyria_mystery
				set_variable = {
					name = zz_magic_teacher
					value = scope:zz_magic_teacher
				}
			}
			scope:zz_magic_teacher = {
				every_memory = {
					limit = { 
						has_memory_category = magic
					}
					root = {
						add_to_variable_list = {
							name = zz_valyria_spell_list
							target = prev
						}
						if = {
							limit = {
								NOT = { has_variable = zz_valyria_mystery }
							}
							set_variable = { name = zz_valyria_mystery value = prev }
						}
					}
				}
			}
		}
	}
}
