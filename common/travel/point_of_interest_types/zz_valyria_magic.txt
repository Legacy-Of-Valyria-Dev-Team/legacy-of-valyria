poi_valyria_magic = {
	build_province_list = {
		province:8126 = {
			add_to_list = provinces
		}

		every_province = { 
			if = { 
				limit = {
					location = { 
						zz_has_natural_dragon_pit_county = yes 
					}
				}
				add_to_list = provinces
			}
		}

		every_province = {	
			limit = {
				has_province_modifier = travel_point_of_interest_magic
			}
			add_to_list = provinces
		}

		every_special_building_province = {	
			limit = {
				has_building_with_flag = { flag = travel_point_of_interest_magic }
			}
			add_to_list = provinces
		}
	}
	
	

	on_visit = {
		send_interface_toast = {
			title = poi_magic.visit
			left_icon = root
			add_character_flag = dragon_training
			zz_random_gain_magic_spell = yes 
			traveler_travel_xp_effect = {
				MIN = 3
				MAX = 5
			}
		}
	}
}
