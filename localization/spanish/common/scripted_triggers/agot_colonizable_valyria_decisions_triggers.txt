agot_can_use_dragonpit_amenities = {
	any_held_county = {
		OR = {
			any_county_province = {
				has_holding_type = castle_holding
				has_building_or_higher = generic_dragon_pit_01
			}
			any_county_province = {
				OR = {
					zz_has_dragon_pit_building = yes
				}
			}
		}
		has_variable = has_dragonkeeper_order
	}
}
