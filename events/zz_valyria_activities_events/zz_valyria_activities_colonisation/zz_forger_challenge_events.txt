zz_valyria_colonisation.0300 = {
	type = activity_event
	title = zz_valyria_colonisation.0300.t
	desc = zz_valyria_colonisation.0300.desc
	theme = university

	left_portrait = {
		character = root
		animation = personality_rational
	}

	cooldown = { years = 7 } ## Changed the Cooldown to 7 years. This is to stop this firing every single time

	trigger = {
		always = yes
	}

	weight_multiplier = {
		base = 1
		modifier = {
			has_trait = theologian
			add = 2
		}
		modifier = {
			has_trait = scholar
			add = 1
		}
	}

	immediate = {
		scope:activity = {
			activity_location = { save_scope_as = location }
		}
		root = { save_scope_as = book_creator }
	}

	# Option A: Investigate documents yourself
	option = {
		name = zz_valyria_colonisation.0300.a
		trigger = {
		}
		duel = {
			skill = diplomacy
			value = high_skill_rating
			65 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				custom_tooltip = zz_valyria_colonisation.0300.a.success.tt
				trigger_event = zz_valyria_colonisation.03001
				stress_impact = {
					base = medium_stress_impact_loss
				}
			}
			35 = {
				custom_tooltip = zz_valyria_colonisation.0300.a.failure.tt
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				trigger_event = zz_valyria_colonisation.03002
				stress_impact = {
					base = medium_stress_impact_gain
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
	
	# Option B: Make him disappear
	option = {
		name = zz_valyria_colonisation.0300.b
		trigger = {
		}
		duel = {
			skill = intrigue
			value = high_skill_rating
			65 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				custom_tooltip = zz_valyria_colonisation.0300.b.success.tt
				stress_impact = {
					base = medium_stress_impact_loss
				}
				hidden_effect = {
					create_character = {
						gender = male
						culture = root.culture
						faith = root.faith
						dynasty = none
						random_traits = yes

						martial = 5
						diplomacy = 5
						intrigue = 5
						stewardship = 5
						learning = 5
						prowess = 5
						
						trait = ambitious
						trait = deceitful
						trait = education_intrigue_2

						save_scope_as = forger
					}
					scope:forger = {
						death = {
							killer = root
							death_reason = death_disappearance
						}
					}
				}
				
			}
			35 = {
				custom_tooltip = zz_valyria_colonisation.0300.b.failure.tt
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				hidden_effect = {
					create_character = {
						gender = male
						culture = root.culture
						faith = root.faith
						dynasty = none
						random_traits = yes

						martial = 5
						diplomacy = 5
						intrigue = 5
						stewardship = 5
						learning = 5
						prowess = 5
						
						trait = ambitious
						trait = deceitful
						trait = education_intrigue_2

						save_scope_as = forger
					}
					scope:forger = {
						death = {
							killer = root
							death_reason = death_murder
						}
					}
					
					root = {
						add_trait = murderer
						add_tyranny = major_tyranny_value
					}
				}
				stress_impact = {
					base = medium_stress_impact_gain
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}

	# Option C: Pay him off!
	option = {
		name = zz_valyria_colonisation.0300.c
		scope:activity = { activity_special_type_progression_medium = yes }
		stress_impact = {
			base = medium_stress_impact_loss
		}
		remove_short_term_gold = {
			gold = {
				value = root.medium_gold_value_check
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				has_trait = lazy
				add = 100
			}
			modifier = {
				has_trait = intellect_bad
				add = 100
			}
			modifier = {
				has_trait = dull
				add = 100
			}
		}
	}
}
