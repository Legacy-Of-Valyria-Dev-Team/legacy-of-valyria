gogossos_expedition.0002 = {
    type = character_event
    title = gogossos_expedition_0002_t
    desc = gogossos_expedition_0002_desc

    theme = hunting
    override_background = wilderness_forest

    left_portrait = {
		character = root
		animation = survey
	}

    immediate = {
		if = {
			limit = {
				NOR = {
					has_variable = pillar_one
					has_variable = pillar_two
					has_variable = pillar_three
				}
			}
			set_variable = {
				name = pillar_one
				value = 0
			}
			set_variable = {
				name = pillar_two
				value = 1
			}
			set_variable = {
				name = pillar_three
				value = 2
			}
		}
    }

	option = {
		name = BYEEEE

		if = {
			limit = {
				var:pillar_one = 4
				var:pillar_one = 2
				var:pillar_one = 1
			}
            remove_variable = pillar_one
		    remove_variable = pillar_two
		    remove_variable = pillar_three
			random = {
                50 = {
                    trigger_event = gogossos_expedition.0003a
                }
                
                50 = {
                    trigger_event = gogossos_expedition.0003b
                }
            }
		}
		else = {
			if = {
                limit = {
                    AND = {
                        NOT = {
                            has_trait = wounded_1
                        }
                        NOR = {
                            has_trait = wounded_2
                            has_trait = wounded_3
                        }
                    }
                }
                add_trait = wounded_1
            }
            else = {
                death = yes
            }
            trigger_event = gogossos_expedition.0002
		}
    }

    option = {
		name = pillar_mural_one
		if = {
			limit = {
				var:pillar_one >= 3
			}
			set_variable = {
				name = pillar_one
				value = 0
			}
		}
		else = {
			change_variable = {
				name = pillar_one
				add = 1
			}
		}
		trigger_event = gogossos_expedition.0002
    }

    option = {
		name = pillar_mural_two
		if = {
			limit = {
				var:pillar_two >= 3
			}
			set_variable = {
				name = pillar_two
				value = 0
			}
		}
		else = {
			change_variable = {
				name = pillar_two
				add = 1
			}
		}
		trigger_event = gogossos_expedition.0002
    }

	option = {
		name = pillar_mural_three
		if = {
			limit = {
				var:pillar_three >= 3
			}
			set_variable = {
				name = pillar_three
				value = 0
			}
		}
		else = {
			change_variable = {
				name = pillar_three
				add = 1
			}
		}
		trigger_event = gogossos_expedition.0002
    }
}
