lv_scenario_cob.0005 = { #Aurion Varezys
	type = character_event
	title = lv_scenario_cob.0005.t
	desc = lv_scenario_cob.0005.desc

	override_background = {
		reference = ep3_campfire
	}
	theme = landless_adventurer
	left_portrait = root
	lower_center_portrait = character:Targaryen_110
	right_portrait = character:Targaryen_122

	immediate = {
		play_music_cue = mood_the_hedge_knight
		character:Targaryen_110 = {
			save_scope_as = maekar
		}
		character:Targaryen_122 = {
			save_scope_as = aegonfive
		}
		character:Tall_1 = {
			save_scope_as = duncan
		}
		hidden_effect = {
			if = {
				limit = {
					scope:aegonfive = {
						NOT = { is_courtier_of = scope:duncan }
					}
				}
				scope:duncan = {
					add_courtier = scope:aegonfive
					scope:aegonfive = {
						return_to_court = yes
					}
				}
			}
			add_character_modifier = { modifier = laamp_provisions_forgiveness_travel_modifier years = 1 }
		}
	}

    # To Dorne, the kin of my squire
	option = {
		name = lv_scenario_cob.0005.a
		custom_tooltip = lv_scenario_cob.0005.a_tt
		add_prestige = 100
		domicile = {
			if = {
				limit = { provisions < max_provisions }
				change_provisions = {
					value = 1000
				}
			}
		}		
		title:k_dorne.holder.capital_province = {
			save_scope_as = relocation_destination
		}
		start_travel_plan = {
			destination = scope:relocation_destination
			on_start_on_action = on_travel_relocation_start
			on_arrival_on_action = on_travel_relocation_end
			on_arrival_destinations = last
			return_trip = no # One way
			travel_with_domicile = yes
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_historic_chain_outcome
			}
		}
	}
    # I will decide my own path
	option = {
		name = lv_scenario_cob.0005.b
		add_prestige = 100
		domicile = {
			if = {
				limit = { provisions < max_provisions }
				change_provisions = {
					value = 1000
				}
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_nonhistoric_chain_outcome
			}
		}
	}
}
